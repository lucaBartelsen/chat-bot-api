API Usage Example

fetch('https://chatsassistant.com/api/suggestions', {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json',
        'Authorization': 'Bearer YOUR_TOKEN'
    },
    body: JSON.stringify({
        message: "Hello, how are you today?",
        chat_history: [],
        regenerate: false
    })
})
.then(response => response.json())
.then(data => console.log(data));
